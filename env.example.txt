# -------------------------------------------
# Discord Configuration
# -------------------------------------------
# Bot token from Discord Developer Portal
# Get it at: https://discord.com/developers/applications
# Go to your app -> Bot -> Reset Token -> Copy
DISCORD_TOKEN=your_discord_bot_token_here

# Application ID (Client ID) for slash commands
# Get it at: https://discord.com/developers/applications
# Go to your app -> General Information -> Application ID
DISCORD_CLIENT_ID=your_discord_client_id_here

# Optional: Guild ID for development (faster command registration)
# Leave empty for global commands (takes up to 1 hour to propagate)
# Right-click your server -> Copy Server ID (Developer Mode must be ON)
DISCORD_GUILD_ID=

# -------------------------------------------
# Supabase (Postgres) Configuration
# -------------------------------------------
# PostgreSQL connection string from Supabase
# Get it at: https://supabase.com/dashboard
# Go to your project -> Settings -> Database -> Connection string (URI)
# IMPORTANT: Use the "Transaction" or "Session" pooler URL for best results
# Format: postgresql://postgres.[project-ref]:[password]@aws-0-[region].pooler.supabase.com:6543/postgres
SUPABASE_DB_URL=postgresql://postgres:your_password@db.your-project-ref.supabase.co:5432/postgres

# -------------------------------------------
# Logging Configuration

# Log level: debug, info, warn, error
